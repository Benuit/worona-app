$(function(){
	$('a').each(function(){
				var href = $(this).attr('href');
	           	$(this).attr('href', 'javascript:');
	           	$(this).click( function(e) {
	           		e.preventDefault();
                    var xobj = new XMLHttpRequest();
                    xobj.open('GET', "http://127.0.0.1:8087?url=" + href + "&", true);
                    xobj.send(null);
	           	});
	     });
	//update();
});
